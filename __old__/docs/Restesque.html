function Route(options) {
    this.path = options.path;
    this.router = options.router;

    this.action = null;
    this.context = null;
    this.params = {};
}

Route.prototype.to = function (fn, context) {
    this.context = (typeof context === 'undefined') ? this : context;
    this.action = fn;
    return this.router;
};

Route.prototype.run = function (arr) {
    this.router.routes[this.path].action.apply(this.contect, [this.router.routes[this.path].params].concat( arr ) );
};
